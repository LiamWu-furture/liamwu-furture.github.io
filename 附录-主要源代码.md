## 一、共享播放引擎 js/player.js

该模块是整个项目的核心，实现了跨页面续播功能。通过 localStorage 持久化播放状态，暴露全局 NeteasePlayer 接口供各页面调用。

```javascript
/* 网易云音乐仿站 - 共享播放引擎
 * 跨页面续播：歌曲、进度、音量、静音、播放模式、喜欢状态全部经 localStorage 持久化。
 * 暴露全局 window.NeteasePlayer，页面脚本通过它驱动 UI 并订阅变化。
 */
(function () {
    'use strict';

    // 核心歌单（5首真实音源，全站唯一数据源）
    const CORE_PLAYLIST = [
        {
            name: "你曾是少年", artist: "焦迈奇", album: "你曾是少年",
            src: "audio/你曾是少年.mp3",
            cover: "images/songs/shaonian.jpg",
            duration: "04:23",
            bg: "linear-gradient(135deg, #3a7bd5, #3a6073)",
            lyrics: ["许多年前", "你曾是个朴素的少年", "爱上一个人", "就不怕付出自己一生", "相信爱会永恒"]
        },
        // ... 其余4首歌曲数据省略 ...
    ];

    // 播放模式枚举
    const MODE = { ORDER: 0, SHUFFLE: 1, REPEAT_ONE: 2 };
    const STORAGE_KEY = 'nem_player';

    // 状态管理：从 localStorage 读取，提供默认值
    function loadState() {
        let s = {};
        try { s = JSON.parse(localStorage.getItem(STORAGE_KEY)) || {}; } catch (e) { s = {}; }
        return {
            index: Number.isInteger(s.index) ? s.index : 0,
            position: typeof s.position === 'number' ? s.position : 0,
            playing: !!s.playing,
            volume: typeof s.volume === 'number' ? s.volume : 0.7,
            muted: !!s.muted,
            mode: [0, 1, 2].includes(s.mode) ? s.mode : MODE.ORDER,
            liked: (s.liked && typeof s.liked === 'object') ? s.liked : {}
        };
    }

    const state = loadState();

    // 持久化状态到 localStorage
    function saveState() {
        try {
            localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
        } catch (e) { /* 隐私模式或 file:// 限制，忽略 */ }
    }

    // 单例 Audio 对象
    const audio = new Audio();
    audio.preload = 'auto';
    audio.volume = state.muted ? 0 : state.volume;
    audio.muted = state.muted;

    // 订阅者模式：页面 UI 通过 onChange 注册回调
    const subscribers = [];
    const progressSubscribers = [];

    function notify() {
        const snapshot = getSnapshot();
        subscribers.forEach(function(cb) { try { cb(snapshot); } catch (e) {} });
    }

    function getSnapshot() {
        return {
            index: state.index,
            song: CORE_PLAYLIST[state.index],
            playing: !audio.paused && !audio.ended,
            mode: state.mode,
            muted: state.muted,
            volume: state.volume,
            liked: isLiked(CORE_PLAYLIST[state.index])
        };
    }

    // 核心播放逻辑
    function loadSong(index, autoplay) {
        state.index = ((index % CORE_PLAYLIST.length) + CORE_PLAYLIST.length) % CORE_PLAYLIST.length;
        var song = CORE_PLAYLIST[state.index];
        if (audio.src !== new URL(song.src, location.href).href) {
            audio.src = song.src;
        }
        if (autoplay) {
            var p = audio.play();
            if (p && p.catch) {
                p.then(function() { state.playing = true; saveState(); })
                 .catch(function() { state.playing = false; armResumeOnInteract(); saveState(); });
            }
        }
        saveState();
        notify();
    }

    function playByIndex(index) { loadSong(index, true); }

    function toggle() {
        if (audio.paused) { var p = audio.play(); if (p && p.catch) p.catch(function() {}); }
        else { audio.pause(); }
    }

    function next(userInitiated) {
        if (state.mode === MODE.SHUFFLE) {
            var n = state.index;
            if (CORE_PLAYLIST.length > 1) {
                while (n === state.index) n = Math.floor(Math.random() * CORE_PLAYLIST.length);
            }
            loadSong(n, true);
        } else { loadSong(state.index + 1, true); }
    }

    function prev() {
        if (state.mode === MODE.SHUFFLE) { next(); return; }
        loadSong(state.index - 1, true);
    }

    function seek(pct) {
        if (audio.duration) audio.currentTime = Math.min(Math.max(pct, 0), 1) * audio.duration;
    }

    function setVolume(v) {
        v = Math.min(Math.max(v, 0), 1);
        state.volume = v; state.muted = v === 0;
        audio.muted = state.muted; audio.volume = v;
        saveState(); notify();
    }

    function toggleMute() {
        state.muted = !state.muted;
        audio.muted = state.muted;
        audio.volume = state.muted ? 0 : state.volume;
        saveState(); notify();
    }

    function cycleMode() {
        state.mode = (state.mode + 1) % 3;
        audio.loop = false;
        saveState(); notify();
    }

    function toggleLike() {
        var song = CORE_PLAYLIST[state.index];
        if (!song) return;
        if (state.liked[song.name]) delete state.liked[song.name];
        else state.liked[song.name] = true;
        saveState(); notify();
    }

    // 页面隐藏或卸载时立即落盘，保证跨页续播精确
    function persistNow() { state.position = audio.currentTime; saveState(); }
    window.addEventListener('pagehide', persistNow);
    document.addEventListener('visibilitychange', function() { if (document.hidden) persistNow(); });

    // 初始恢复：页面加载时立即执行
    (function restore() {
        var song = CORE_PLAYLIST[state.index];
        audio.src = song.src;
        audio.muted = state.muted;
        audio.volume = state.muted ? 0 : state.volume;
        audio.addEventListener('loadedmetadata', function once() {
            audio.removeEventListener('loadedmetadata', once);
            if (state.position && isFinite(state.position) && state.position < audio.duration) {
                try { audio.currentTime = state.position; } catch (e) {}
            }
        });
        if (state.playing) {
            var p = audio.play();
            if (p && p.catch) p.catch(function() { armResumeOnInteract(); });
        }
    })();

    // 公开 API
    window.NeteasePlayer = {
        MODE: MODE,
        playlist: CORE_PLAYLIST,
        playByIndex: playByIndex,
        toggle: toggle,
        next: function() { next(true); },
        prev: prev,
        seek: seek,
        setVolume: setVolume,
        toggleMute: toggleMute,
        cycleMode: cycleMode,
        toggleLike: toggleLike,
        formatTime: formatTime,
        isLiked: isLiked,
        getState: getSnapshot,
        indexOfName: function(name) {
            return CORE_PLAYLIST.findIndex(function(s) {
                return s.name === (SONG_NAME_ALIASES[name] || name);
            });
        },
        coverStyle: coverStyle,
        onChange: function(cb) { if (typeof cb === 'function') { subscribers.push(cb); cb(getSnapshot()); } },
        onProgress: function(cb) { if (typeof cb === 'function') { progressSubscribers.push(cb); cb(audio.currentTime, audio.duration || 0); } }
    };
})();
```

## 五、跨页面续播实现原理

本项目的核心创新点是跨页面续播功能，其实现原理如下：

**1. 状态持久化**

播放状态（当前歌曲索引、播放进度、音量、是否静音、播放模式、喜欢列表）通过 localStorage 持久化存储。

```javascript
// 存储键名
var STORAGE_KEY = 'nem_player';

// 保存状态
function saveState() {
    try {
        localStorage.setItem(STORAGE_KEY, JSON.stringify(state));
    } catch (e) {
        // 隐私模式或 file:// 协议限制，忽略错误
    }
}
```

**2. 页面加载时恢复**

每个页面加载时，player.js 会自动从 localStorage 读取上次的播放状态，并恢复到对应位置。

```javascript
// 初始恢复
(function restore() {
    var song = CORE_PLAYLIST[state.index];
    audio.src = song.src;
    audio.muted = state.muted;
    audio.volume = state.muted ? 0 : state.volume;

    // 等待音频元数据加载完成后，跳转到上次的播放位置
    audio.addEventListener('loadedmetadata', function once() {
        audio.removeEventListener('loadedmetadata', once);
        if (state.position && isFinite(state.position) && state.position < audio.duration) {
            try { audio.currentTime = state.position; } catch (e) {}
        }
    });

    // 如果上次是播放状态，自动继续播放
    if (state.playing) {
        var p = audio.play();
        if (p && p.catch) p.catch(function() { armResumeOnInteract(); });
    }
})();
```

**3. 实时落盘**

播放过程中，每隔约1秒自动保存一次进度；页面隐藏或卸载时立即保存，确保进度精确。

```javascript
// timeupdate 事件中节流保存
var lastSaved = 0;
audio.addEventListener('timeupdate', function() {
    state.position = audio.currentTime;
    var now = Date.now();
    if (now - lastSaved > 1000) {
        lastSaved = now;
        saveState();
    }
});

// 页面隐藏时立即落盘
window.addEventListener('pagehide', function() {
    state.position = audio.currentTime;
    saveState();
});
```
